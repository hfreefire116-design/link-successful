<script>
//<![CDATA[

/**
 * UNLOCK PAGE - TASK VERIFICATION
 * ================================
 * 
 * Purpose:
 * Displays and manages task completion for content unlock.
 * Users must complete required tasks (subscribe, follow, etc.)
 * before gaining access to destination content.
 * 
 * Functionality:
 * - Decrypts link configuration from URL
 * - Displays tasks with progress tracking
 * - Manages task verification timers
 * - Handles link opening securely
 * - Updates UI based on completion status
 * 
 * Security:
 * - AES-256 decryption of encrypted data
 * - URL parameter sanitization
 * - Session-based state (no localStorage for sensitive data)
 * - Secure window.open with noopener, noreferrer
 * 
 * No data sent to final page - everything
 * handled client-side only
 */

const SECRET_KEY = "566494444";
let data = null;
let currentIdx = 0;
let isTimerRunning = false;

function safeCreateIcons() {
  try {
    if (window.lucide && typeof window.lucide.createIcons === 'function') {
      window.lucide.createIcons();
    }
  } catch(e) {
    console.warn('Icon init failed:', e);
  }
}

window.addEventListener('DOMContentLoaded', safeCreateIcons);

window.addEventListener('error', (e) => {
  console.error('Error:', e);
  showErrorState();
});

/**
 * Initializes page with decrypted data from URL
 */
window.onload = () => {
  try {
    const params = new URLSearchParams(window.location.search);
    const raw = params.get('data');
    
    if (raw) {
      const fixed = raw.replace(/ /g, '+');
      const decrypted = CryptoJS.AES.decrypt(fixed, SECRET_KEY).toString(CryptoJS.enc.Utf8);
      
      if(!decrypted) throw new Error("Decryption failed");
      
      data = JSON.parse(decrypted);
      
      // Clean URL to hide encrypted data
      const cleanURL = window.location.origin + window.location.pathname;
      window.history.replaceState({}, document.title, cleanURL);
      
      if(!data.tasks || data.tasks.length === 0) {
        showErrorState("No tasks configured");
      } else {
        renderAllTasks();
      }
    } else {
      showErrorState("Invalid link");
    }
  } catch(e) {
    console.error('Init error:', e);
    showErrorState();
  }
};

/**
 * Displays error state
 */
function showErrorState(msg = "Invalid or expired link") {
  document.getElementById('unlock-btn').textContent = msg;
  document.getElementById('unlock-btn').disabled = true;
  document.getElementById('task-list').innerHTML = '';
}

/**
 * Maps icon names to lucide icons
 */
function mapIcon(name) {
  const map = {
    'link': 'link',
    'youtube': 'youtube',
    'facebook': 'facebook',
    'instagram': 'instagram',
    'twitter': 'twitter',
    'discord': 'gamepad-2',
    'telegram': 'send',
    'tiktok': 'music'
  };
  return map[name] || 'link';
}

/**
 * Renders all tasks to the page
 */
function renderAllTasks() {
  const container = document.getElementById('task-list');
  const btn = document.getElementById('unlock-btn');
  const progress = document.getElementById('progress');
  
  if(!container) return;

  const isFinished = currentIdx >= data.tasks.length;
  
  // Update progress bar
  const percent = (currentIdx / data.tasks.length) * 100;
  progress.style.width = `${percent}%`;
  
  // Update button
  if (isFinished) {
    btn.classList.remove('disabled');
    btn.classList.add('active');
    document.getElementById('btn-text').textContent = 'UNLOCK CONTENT';
  } else {
    btn.classList.add('disabled');
    btn.classList.remove('active');
    document.getElementById('btn-text').textContent = `COMPLETE ALL TASKS (${currentIdx}/${data.tasks.length})`;
  }

  // Render tasks
  let html = '';
  data.tasks.forEach((task, idx) => {
    const isCompleted = idx < currentIdx;
    const isCurrent = idx === currentIdx;
    const isLocked = idx > currentIdx;
    
    let classes = 'task-item';
    if (isCompleted) classes += ' completed';
    if (isCurrent) classes += ' active';
    if (isLocked) classes += ' locked';
    
    const iconName = mapIcon(task.icon);
    let actionHtml = '';
    
    if (isCurrent) {
      if (isTimerRunning) {
        actionHtml = '<div class="task-action"><div style="width:24px;height:24px;border:2px solid #000;border-top:2px solid transparent;border-radius:50%;animation:spin 1s linear infinite"></div></div>';
      } else {
        actionHtml = '<div class="task-action" style="cursor:pointer;font-size:20px">→</div>';
      }
    }
    
    html += `
      <div class="${classes}" ${isCurrent && !isTimerRunning ? `onclick="handleTaskClick(${idx})"` : ''}>
        <div class="task-icon">
          <i data-lucide="${iconName}" style="width:24px;height:24px;"></i>
        </div>
        <div class="task-content">
          <div class="task-title">${task.name || 'Click Here'}</div>
          <div class="task-subtitle">
            ${isCompleted ? '✓ Completed' : isCurrent ? (isTimerRunning ? 'Verifying...' : 'Tap to open') : 'Locked'}
          </div>
        </div>
        ${actionHtml}
      </div>
    `;
  });
  
  container.innerHTML = html;
  safeCreateIcons();
}

/**
 * Handles task click - opens link and starts timer
 */
function handleTaskClick(idx) {
  if(isTimerRunning) return;
  
  const task = data.tasks[idx];
  if(!task) return;
  
  // Open link in new window
  try {
    window.open(task.url, '_blank', 'noopener,noreferrer');
  } catch(e) {
    console.warn('Open failed:', e);
  }
  
  // Start verification timer
  isTimerRunning = true;
  let timeLeft = parseInt(task.timer || 5, 10);
  renderAllTasks();
  
  const timerId = setInterval(() => {
    timeLeft--;
    if (timeLeft <= 0) {
      clearInterval(timerId);
      isTimerRunning = false;
      currentIdx++;
      
      if (currentIdx >= data.tasks.length) {
        renderAllTasks();
      } else {
        renderAllTasks();
      }
    }
  }, 1000);
}

/**
 * Redirects to destination content
 */
function unlockContent() {
  if (currentIdx < data.tasks.length) return;
  if (!data || !data.dest) {
    alert('Error: No destination URL');
    return;
  }
  window.location.href = data.dest;
}
